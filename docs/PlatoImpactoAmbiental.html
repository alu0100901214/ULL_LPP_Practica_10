def indiceGEI
	totalGEI = totalGEIanual
	if(totalGEI < 800)
		1
	elsif((totalGEI >= 800) && (totalGEI <= 1200))
		2
	elsif(totalGEI > 1200)
		3
	end
end

def totalGEI
	node = @platoList.head
	total_gei = 0
	while(node != nil)
		total_gei += node.value.gei
		node = node.next
	end
	(total_gei).round(2)
end
