def push_head(val)
	if(is_empty)
		node = Node.new(val,nil,nil)
		@head = node
		@tail = node
	end

	if(!is_empty)
		node = Node.new(val,@head,nil)
		@head.prev = node
		@head = node
	end
	@size += 1
end

def push_tail(val)
	if(is_empty)
		node = Node.new(val,nil,nil)
		@head = node
		@tail = node
	end
	if(!is_empty)
		node = Node.new(val,nil,@tail)
		@tail.next = node
		@tail = node
	end
	@size += 1
end
